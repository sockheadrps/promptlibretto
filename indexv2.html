The Big Picture

                
promptlibretto assembles prompts from composable pieces — like a musical score built from parts.

                
Instead of one monolithic prompt string, you define routes, context layers, and injections that snap together at generation time.

                
Once you've tuned and tested a configuration in the studio, export it as JSON. In your project, a single call loads the full prompt pipeline — context, route, injections, sampling, and output policy — ready to run:

                
from promptlibretto import load_engine
  
  eng = load_engine("analyst_v2.json")
  result = eng.run(input="How do I get my first 10 paying users?")

                
Step through to see each piece add to the prompt on the right →

              

  
              

                
Base Context & Route

                
Base Context is your long-lived framing — persona, constraints, stable facts. It appears in every prompt.

                
A route is a named strategy for one kind of prompt. It bundles a composition recipe, sampling defaults, and an output policy. A route can define a runtime schema to create structured inputs.

              

  
              

                
Overlays

                
Overlays are temporary layers stacked on top of the base context — transient facts, signals, or emphasis. Each has a name and priority (higher = appears first). Some are fixed, others are filled at runtime via templates.

              

  
              

                
Injections

                
Injections are reusable fragments that modify the active route — adding instruction text, few-shot examples, and overriding sampling or output policy. Use them to represent conditional logic or stylistic variants.

              

  
              

                
Generation & Output

                
When you hit Generate, the engine assembles all the active layers, sends the final prompt to the model, and validates the response. Enable Debug Trace to inspect the exact prompts and attempts.

              

  
              

                
Pre-Generate & Export

                
Pre-Generate previews assembled sections before sending. Export saves the entire configuration as JSON, loadable with promptlibretto.load_engine(). Runtime overlays and schema fields become function arguments.

              

  
              

                
Ensemble

                
The Ensemble page fans out the same input across multiple saved exports, each running independently with its own route, context, and config. Compare results side-by-side to find the best model for the job.

document.addEventListener('DOMContentLoaded', () => {
  // --- STATE MANAGEMENT ---
  let state = {
    baseContext: '',
    selectedRoute: 'auto-route',
    overlays: [], // { name, text, priority, runtime, template }
    activeInjections: new Set(),
    generationOverrides: {},
    userInput: {}, // Will hold values from dynamic inputs
    // Data loaded from backend
    routes: [],
    injections: [],
    personas: [],
    examplePools: {},
  };

  // --- DOM ELEMENT REFERENCES ---
  const $ = (selector) => document.querySelector(selector);
  const $$ = (selector) => document.querySelectorAll(selector);

  // Tabs
  const composeTab = $('[data-tab="compose"]');
  const tuningTab = $('[data-tab="state"]');
  const composePanel = $('.tab-panel[data-tab="compose"]');
  const tuningPanel = $('.tab-panel[data-tab="state"]');
  const outputTab = $('[data-output-tab="output"]');
  const stageTab = $('[data-output-tab="stage"]');
  const outputPanel = $('.output-panel-body[data-output-tab="output"]');
  const stagePanel = $('.output-panel-body[data-output-tab="stage"]');

  // Main Controls
  const baseText = $('#base-text');
  const routeSelect = $('#route-select');
  const generateBtn = $('#generate-btn');
  const exportBtn = $('#export-btn');
  
  // Input Container
  const inputContainer = $('#input-container');
  const defaultInputGroup = $('#input-group-default');
  const defaultInputText = $('#input-text');

  // Modals
  const customRouteModal = $('#custom-route-modal');
  const injectionModal = $('#injection-modal');
  const aboutModal = $('#about-modal');

  // --- DUMMY API CALLS ---
  const api = {
    // GET requests to fetch initial data
    getRoutes: () => fetch('/api/dummy/routes').then(res => res.json()),
    getInjections: () => fetch('/api/dummy/injections').then(res => res.json()),
    getPersonas: () => fetch('/api/dummy/personas').then(res => res.json()),
    getExamplePools: () => fetch('/api/dummy/example-pools').then(res => res.json()),
    
    // POST requests to create/update data
    saveRoute: (routeData) => fetch('/api/dummy/routes', { method: 'POST', body: JSON.stringify(routeData) }),
    saveInjection: (injData) => fetch('/api/dummy/injections', { method: 'POST', body: JSON.stringify(injData) }),
    savePersona: (personaData) => fetch('/api/dummy/personas', { method: 'POST', body: JSON.stringify(personaData) }),
    saveExamplePool: (poolData) => fetch('/api/dummy/example-pools', { method: 'POST', body: JSON.stringify(poolData) }),
    
    // Main generation and export calls
    generate: (requestPayload) => fetch('/api/dummy/generate', { method: 'POST', body: JSON.stringify(requestPayload) }).then(res => res.json()),
    export: (exportPayload) => fetch('/api/dummy/export', { method: 'POST', body: JSON.stringify(exportPayload) }).then(res => res.json()),
  };
  
  // --- INITIALIZATION ---
  const initializeApp = async () => {
    // Mocking API responses since we don't have a backend
    mockApiResponses();

    // Fetch all initial data in parallel
    const [routes, injections, personas, examplePools] = await Promise.all([
      api.getRoutes(),
      api.getInjections(),
      api.getPersonas(),
      api.getExamplePools(),
    ]);

    state.routes = routes;
    state.injections = injections;
    state.personas = personas;
    state.examplePools = examplePools;

    // Populate UI with fetched data
    populateRouteSelect();
    renderInjections();
    renderPersonas();
    
    // Set up event listeners
    setupEventListeners();

    console.log('Prompt Studio Initialized with state:', state);
  };

  // --- UI RENDERING & UPDATES ---

  const populateRouteSelect = () => {
    routeSelect.innerHTML = '';
    state.routes.forEach(route => {
      const option = document.createElement('option');
      option.value = route.name;
      option.textContent = route.name;
      routeSelect.appendChild(option);
    });
  };

  const renderInjections = () => {
    const container = $('#injection-groups');
    container.innerHTML = '';
    state.injections.forEach(inj => {
        const label = document.createElement('label');
        label.className = 'switch';
        label.innerHTML = `
            
            
            ${inj.name}
        `;
        container.appendChild(label);
    });
  };
  
  const renderPersonas = () => {
      const list = $('#persona-list');
      list.innerHTML = '';
      state.personas.forEach(p => {
          const div = document.createElement('div');
          div.className = 'overlay-item'; // Reusing style
          div.textContent = p.persona;
          div.onclick = () => {
              $('#persona-text').value = p.persona;
              $('#persona-nudges').value = p.nudges.join('\n');
          };
          list.appendChild(div);
      });
  };

  const updateRouteUI = () => {
    const routeName = routeSelect.value;
    const route = state.routes.find(r => r.name === routeName);

    // Hide all special input groups
    $$('.input-group-special').forEach(el => el.hidden = true);
    defaultInputGroup.hidden = false;
    
    if (!route || !route.schema) {
      $('#route-desc').textContent = route ? route.description : 'Automatically select the best route.';
      return;
    }

    $('#route-desc').textContent = route.description;

    // Show special input group for this route
    let inputGroup = $(`#input-group-${route.name}`);
    if (!inputGroup) {
      inputGroup = document.createElement('div');
      inputGroup.id = `input-group-${route.name}`;
      inputGroup.className = 'input-group-special';
      
      route.schema.forEach(field => {
        const label = document.createElement('label');
        const span = document.createElement('span');
        span.textContent = field.name;
        label.appendChild(span);
        
        if (field.type === 'select' && field.options) {
            const select = document.createElement('select');
            select.dataset.key = field.name;
            field.options.forEach(opt => {
                const option = document.createElement('option');
                option.value = opt;
                option.textContent = opt;
                select.appendChild(option);
            });
            label.appendChild(select);
        } else {
            const input = document.createElement('input');
            input.type = 'text';
            input.placeholder = field.name;
            input.dataset.key = field.name; // To identify which part of state this maps to
            label.appendChild(input);
        }
        inputGroup.appendChild(label);
      });
      inputContainer.appendChild(inputGroup);
    }
    
    defaultInputGroup.hidden = true;
    inputGroup.hidden = false;
  };
  
  // --- EVENT LISTENERS ---
  const setupEventListeners = () => {
    // Tab controls
    composeTab.addEventListener('click', () => {
      composeTab.classList.add('active');
      tuningTab.classList.remove('active');
      composePanel.hidden = false;
      tuningPanel.hidden = true;
    });
    tuningTab.addEventListener('click', () => {
      tuningTab.classList.add('active');
      composeTab.classList.remove('active');
      tuningPanel.hidden = false;
      composePanel.hidden = true;
    });
    outputTab.addEventListener('click', () => {
        outputTab.classList.add('active');
        stageTab.classList.remove('active');
        outputPanel.hidden = false;
        stagePanel.hidden = true;
    });
    stageTab.addEventListener('click', () => {
        stageTab.classList.add('active');
        outputTab.classList.remove('active');
        stagePanel.hidden = false;
        outputPanel.hidden = true;
    });

    // Main controls
    routeSelect.addEventListener('change', updateRouteUI);
    generateBtn.addEventListener('click', handleGenerate);

    // Modal Triggers
    $('#new-custom-route-btn').addEventListener('click', () => customRouteModal.hidden = false);
    $('[data-close-custom-route]').addEventListener('click', () => customRouteModal.hidden = true);
    $('#new-injection-btn').addEventListener('click', () => injectionModal.hidden = false);
    $('[data-close-injection]').addEventListener('click', () => injectionModal.hidden = true);
    $('#about-btn').addEventListener('click', () => aboutModal.hidden = false);
    $('[data-close-about]').addEventListener('click', () => aboutModal.hidden = true);
    
    // State change listeners
    baseText.addEventListener('input', (e) => state.baseContext = e.target.value);
    $('#injection-groups').addEventListener('change', (e) => {
        if (e.target.type === 'checkbox') {
            const name = e.target.dataset.injectionName;
            if (e.target.checked) {
                state.activeInjections.add(name);
            } else {
                state.activeInjections.delete(name);
            }
        }
    });

    // Persona Management
    $('#add-persona').addEventListener('click', async () => {
        const persona = {
            persona: $('#persona-text').value,
            nudges: $('#persona-nudges').value.split('\n').filter(n => n.trim() !== '')
        };
        if (!persona.persona) return alert('Persona text cannot be empty.');

        await api.savePersona(persona);
        state.personas = await api.getPersonas(); // Re-fetch
        renderPersonas();
        alert('Persona saved!');
    });

    // Example Pool Management
    $('#load-examples').addEventListener('click', () => {
        const poolName = $('#example-pool-select').value;
        const examples = state.examplePools[poolName] || [];
        $('#example-pool-text').value = examples.join('\n');
    });

    $('#save-examples').addEventListener('click', async () => {
        const poolName = $('#example-pool-select').value;
        const examples = $('#example-pool-text').value.split('\n').filter(e => e.trim() !== '');
        
        await api.saveExamplePool({ poolName, examples });
        state.examplePools[poolName] = examples;
        alert(`Example pool '${poolName}' saved!`);
    });
  };
  
  // --- ACTION HANDLERS ---
  const handleGenerate = async () => {
    // 1. Collect all user inputs
    const userInput = {};
    const routeName = routeSelect.value;
    const route = state.routes.find(r => r.name === routeName);

    if (route && route.schema) {
        $$(`#input-group-${routeName} [data-key]`).forEach(input => {
            userInput[input.dataset.key] = input.value;
        });
    } else {
        userInput.input = defaultInputText.value;
    }

    // 2. Assemble the request payload
    const requestPayload = {
      base_context: state.baseContext,
      route: routeName,
      injections: Array.from(state.activeInjections),
      // overlays would be collected here
      user_input: userInput,
      // generation_overrides would be collected here
    };

    // 3. Show request in debug trace
    $('#trace-system').textContent = '...sending...';
    $('#trace-user').textContent = JSON.stringify(requestPayload, null, 2);
    $('#output-rendered').textContent = 'Generating...';
    
    // 4. Call the API
    const result = await api.generate(requestPayload);
    
    // 5. Display the result
    $('#output-rendered').textContent = result.output;
    $('#trace-system').textContent = result.debug.final_system_prompt;
    $('#trace-user').textContent = result.debug.final_user_prompt;
    $('#trace-usage').textContent = `Tokens: ${result.usage.total_tokens}, Time: ${result.usage.total_time}s`;
    
    console.log('Generation successful:', result);
  };
  
  // --- MOCKING (for demonstration without a backend) ---
  const mockApiResponses = () => {
      // Intercept fetch calls to provide dummy data
      const originalFetch = window.fetch;
      window.fetch = (url, options) => {
          console.log(`Intercepted fetch for: ${url}`);
          
          if (url === '/api/dummy/routes') {
              return Promise.resolve(new Response(JSON.stringify([
                  { name: 'general_chat', description: 'A standard chat message.' },
                  { name: 'qa_reaction', description: 'React to a question.', schema: [
                      { name: 'question', type: 'text', required: true },
                      { name: 'asker', type: 'text', required: true },
                      { name: 'answer', type: 'text', required: true },
                      { name: 'tone', type: 'select', options: ['normal', 'roast', 'nice'] }
                  ]}
              ])));
          }

          if (url === '/api/dummy/injections') {
              return Promise.resolve(new Response(JSON.stringify([
                  { name: 'positive_opinion', system_addon: 'Your opinion is very positive.' },
                  { name: 'negative_opinion', system_addon: 'Your opinion is very negative.' },
                  { name: 'use_short_words', system_addon: 'Use only 1-5 words.' },
              ])));
          }

          if (url === '/api/dummy/personas') {
              return Promise.resolve(new Response(JSON.stringify([
                  { persona: 'You just tuned in 10 seconds ago.', nudges: ['Ask a question.', 'Sound confused.'] },
                  { persona: 'You are a lurker who never types.', nudges: ['Be chill.', 'Use internet slang.'] }
              ])));
          }

          if (url === '/api/dummy/example-pools') {
              return Promise.resolve(new Response(JSON.stringify({
                  normal: ['lmao', 'pog', 'bruh', 'yooo'],
                  positive: ['lets gooo', 'W stream', 'love this']
              })));
          }
          
          if (url === '/api/dummy/generate') {
              const body = JSON.parse(options.body);
              return Promise.resolve(new Response(JSON.stringify({
                  output: `This is a dummy response for route: ${body.route}. The user asked about: ${JSON.stringify(body.user_input)}`,
                  debug: {
                      final_system_prompt: `[SYSTEM PROMPT]\nBase: ${body.base_context}\nInjections: ${body.injections.join(', ')}`,
                      final_user_prompt: `[USER PROMPT]\n${JSON.stringify(body.user_input, null, 2)}`
                  },
                  usage: { total_tokens: 42, total_time: 0.5 }
              })));
          }
          
          // For save operations, just return a success response
          if (options && options.method === 'POST') {
              return Promise.resolve(new Response(JSON.stringify({ success: true })));
          }

          // Fallback to original fetch for other requests (e.g., stylesheets)
          return originalFetch(url, options);
      };
  };

  // --- START THE APP ---
  initializeApp();
});
